2. Tahun Kabisat

    STORE "year" with any number

    IF "year" mod 4 = 0 
        IF  "year" mod 100 > 0
            PRINT "tahun kabisat"
        ELSE
            PRINT "bukan tahun kabisat"
        END IF
    END IF

    IF "year" mod 4 = 0
        IF "year" mod 100 = 0
            IF "year" mod 400 = 0
                PRINT "tahun kabisat"
            ELSE
                PRINT "bukan tahun kabisat"
            END IF
        END IF
    END IF


3. Laudry Day

    SET "cloth" to "0"
    SET "limit" to "20"

    WHILE "cloth" < "limit"
        COMPUTE "cloth" + 1
    END WHILE

    IF "cloth" = 20
        SET washing machine to "on"
    DISPLAY "Washing clothes"

4. Periksa Kuku

    SET "kuku" to 1
    
    REPEAT
        DO "pemeriksaan kuku"
    UNTIL 40 siswa
    IF "kuku" > 1
        DO hukum siswa
    ELSE

    END IF
