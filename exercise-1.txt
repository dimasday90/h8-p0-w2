2. Tahun Kabisat

    INPUT "year" with any value

    IF "year" mod 4 = 0 
        IF  "year" mod 100 > 0
            PRINT "tahun kabisat"
        ELSE
            PRINT "bukan tahun kabisat"
        END IF
    END IF

    IF "year" mod 4 = 0
        IF "year" mod 100 = 0
            IF "year" mod 400 = 0
                PRINT "tahun kabisat"
            ELSE
                PRINT "bukan tahun kabisat"
            END IF
        END IF
    END IF


3. Laudry Day

    SET "cloth" to "0"
    SET "limit" to "20"

    WHILE "cloth" < "limit"
        COMPUTE "cloth" + 1
    END WHILE


4. Periksa Kuku

    SET "kuku" to 1
    
    REPEAT
        DO "pemeriksaan kuku"
    UNTIL
    IF "kuku" > 1
        DO hukum siswa
    ELSE
        DO puji siswa
    END IF
